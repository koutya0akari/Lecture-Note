\begin{dfn}
  \quad $G$ : 無向グラフ
  \begin{enumerate}[label=(\arabic*)\, ]
    \item $G$ の頂点 $v$ に接続されている辺の数を $G$ の $v$ における\textbf{次数}と呼び、$\deg_G(v)$ で表す。
    \item $G$ の次数が $1$ である頂点を $G$ の \textbf{葉} という。
  \end{enumerate}
\end{dfn}

\begin{prop}
  \quad $G$ : 無向グラフ\par
  $\sum_{v \in V(G)} \deg_G(v) = 2|E(G)|$ が成り立つ。
\end{prop}

\begin{proof}
  各辺はちょうど2つの頂点に接続されているため、$\sum_{v \in V(G)} \deg_G(v)$ は $G$ の辺の数 $|E(G)|$ の2倍になる。
\end{proof}

\begin{cor}
  \quad $G$ : 無向グラフ\par
  $G$ の頂点の次数の和は偶数である。
\end{cor}

\begin{proof}
  $\sum_{v \in V(G)} \deg_G(v) = 2|E(G)|$ より、$\sum_{v \in V(G)} \deg_G(v)$ は偶数である。
\end{proof}

\begin{dfn}
  $G_1$, $G_2$ : 無向グラフ
  \begin{itemize}
    \item $G_1$ と $G_2$ が \textbf{同一} であるとは、$V(G_1) = V(G_2)$ かつ $E(G_1) = E(G_2)$ であることをいう。
    \item $G_1$ と $G_2$ が \textbf{同型} であるとは、全単射 $f: V(G_1) \to V(G_2)$ が存在し、任意の $u, v \in V(G_1)$ に対して、$uv \in E(G_1)$ であることと $f(u)f(v) \in E(G_2)$ であることが同値であることをいう。
  \end{itemize}
\end{dfn}

\end{ex}

\begin{dfn}
  \quad $G$, $H$ : 無向グラフ
  \begin{itemize}
    \item $V(H) \subseteq V(G)$ かつ $E(H) \subseteq E(G)$ であるとき、$H$ は $G$ の \textbf{部分グラフ} であるといい、$H \subseteq G$ と表す。
    \item 上でさらに $V(H) = V(G)$ であるとき、$H$ は $G$ の \textbf{全域部分グラフ} であるという。
    \item $S \subseteq V(G)$ とし、$V(\langle S \rangle_G) = S$ かつ $E(\langle S \rangle_G) = \{uv \in E(G) \mid u, v \in S\}$ と定めるとき、$\langle S \rangle_G$ を $G$ の $S$ による \textbf{誘導部分グラフ} という。
  \end{itemize}
\end{dfn}

\begin{dfn}
  \quad $G$ : 無向グラフ, \quad $v_1, v_2, \ldots, v_n \in V(G)$
  \begin{enumerate}[label=(\arabic*)\, ]
    \item 任意の $1 \leq i \leq n$ に対して、$v_iv_{i+1} \in E(G)$ であるとき、頂点の列 $\left(v_1, v_2, \ldots, v_n\right)$ を $G$ の長さ $n-1$ の \textbf{walk} という。
  \end{enumerate}
  $W = \left(v_1, v_2, \ldots, v_n\right)$ を $G$ の walk とする。
  \begin{enumerate}[label=(\arabic*)\, , resume]
    \item 任意の $1 \leq i < j \leq n$ に対して、$v_iv_{i+1} \neq v_jv_{j+1}$ であるとき、$W$ を $G$ の \textbf{trail} という。
    \item 任意の $1 \leq i < j \leq n$ に対して、$v_i \neq v_j$ であるとき、$W$ を $G$ の \textbf{path} という。
    \item 閉 walk $W$ が 任意の $1 \leq i < j \leq n$ に対して、$v_iv_{i+1} \neq v_jv_{j+1}$ であるとき、$W$ を $G$ の \textbf{circuit} という。
    \item 閉 walk $W$ が 任意の $1 \leq i < j \leq n$ に対して、$v_i \neq v_j$ であるとき、$W$ を $G$ の \textbf{cycle} という。
  \end{enumerate}
\end{dfn}

\begin{dfn}
  \quad $G$ : 無向グラフ\par
  \begin{enumerate}[label=(\arabic*)\, ]
    \item 任意の2点間に対して、$G$ に path が存在するとき、$G$ を \textbf{連結} であるという。
    \item $G$ が連結でないとき、$G$ を \textbf{非連結} であるという。
    \item $G$ の極大な連結部分グラフを $G$ の \textbf{連結成分} という。
    \item $G$ が連結であって、かつ $G$ に cycle が存在しないとき、$G$ を \textbf{木} であるという。
    \item leaf でない頂点が高々1つしか存在しない tree を \textbf{star} という。
    \item leaf の数が $2$ である tree を \textbf{path} という。
    \item 連結グラフ $G$ の全域部分グラフであって、かつ tree であるものを $G$ の \textbf{全域木} という。
  \end{enumerate}
\end{dfn}

\begin{prob}
  どの2つの頂点も同じ次数をもたないグラフが存在しないことを示せ。
\end{prob}

\begin{proof}
  $G$ を頂点数 $n$ の無向グラフとする。$G$ の頂点の次数は $0, 1, 2, \ldots, n-1$ のいずれかである。しかし、次数が $n-1$ である頂点が存在する場合、次数が $0$ である頂点は存在し得ない。したがって、$G$ の頂点の次数は高々 $n-1$ 種類しか存在しないため、鳩の巣原理により、どの2つの頂点も同じ次数をもたないグラフは存在しない。
\end{proof}

\begin{dfn}
  \quad $G$ : 連結グラフ, \quad $u,v \in V(G)$\par
  \begin{enumerate}[label=(\arabic*)\, ]
    \item $G$ の $u$ から $v$ への path の長さの最小値を $G$ の $u$ と $v$ の \textbf{距離} といい、$\operatorname{dist}_G(u,v)$ で表す。
    \item $G$ の $2$ 点間の距離の最大値を $G$ の \textbf{直径} といい、$\text{diam}(G)$ で表す。
    \item 各 $u$ に対して、$\max_{v \in V(G)} \operatorname{dist}_G(u,v)$ を $G$ の $u$ の \textbf{離心率} といい、$\operatorname{ecc}_G(u)$ で表す。
    \item $G$ の離心率の最小値を $G$ の \textbf{半径} といい、$\rad(G)$ で表す。
    \item $G$ の半径と等しい離心率をもつ頂点全体を $G$ の \textbf{中心} といい、$C(G)$ で表す。
  \end{enumerate}
\end{dfn}

\begin{prop}
  \quad $G$ : 連結グラフ \quad このとき、以下が成り立つ
  \begin{enumerate}

    \item 任意の $u, v \in V(G)$ に対して、$\operatorname{dist}_G(u,v) \leq \max_{w \in V(G)} \operatorname{dist}_G(u,w) + \max_{w \in V(G)} \operatorname{dist}_G(v,w)$ が成り立つ。

\end{prop}

\begin{dfn}

\begin{prop}
  グラフ $G$ が二部グラフであるための必要十分条件は、$G$ が奇 cycle をもたないことである。
\end{prop}

\begin{proof}
  $G$ が部集合 $V_1,V_2$ をもつ二部グラフであるとする。cycle 上では隣り合う頂点が常に異なる部集合に属するため、出発点に戻るには偶数本の辺が必要である。したがって、$G$ は奇 cycle をもたない。

  逆に $G$ が奇 cycle をもたないとする。各連結成分ごとに頂点 $r$ を一つ選び、$r$ からの距離が偶数である頂点全体を $V_0$、奇数である頂点全体を $V_1$ とおく。もし $V_0$ 内の $2$ 頂点 $x,y$ が隣接していれば、$r$ から $x,y$ への最短 path と辺 $xy$ から奇 cycle が得られる。同様に $V_1$ 内にも辺は存在しない。よって各連結成分は二部グラフであり、$G$ 全体も二部グラフである。
\end{proof}

\begin{prop}
  すべての非自明な木は、少なくとも $2$ つの葉をもつ。
\end{prop}

\begin{proof}
  木 $T$ の最長 path を $P=(v_1,\ldots,v_k)$ とする。端点 $v_1$ が葉でないならば、$v_1$ は $v_2$ 以外の頂点 $w$ と隣接する。$w$ が $P$ 上にあれば cycle が生じ、$P$ 上になければ $P$ より長い path が得られる。いずれも矛盾である。よって $v_1$ は葉であり、同様に $v_k$ も葉である。
\end{proof}

\begin{prop}
  グラフ $G$ が木であるための必要十分条件は、任意の $2$ 頂点間に唯一の path が存在することである。
\end{prop}

\begin{proof}
  $G$ が木ならば連結であるから、任意の $2$ 頂点間に少なくとも $1$ 本の path が存在する。もし異なる $2$ 本の path が存在すれば、それらの差から cycle が得られるため、木であることに反する。よって path は唯一である。

  逆に任意の $2$ 頂点間に唯一の path が存在するとする。このとき $G$ は連結である。また cycle が存在すれば、その cycle 上の異なる $2$ 頂点間に $2$ 本の path が存在して矛盾する。したがって $G$ は連結かつ cycle をもたないので木である。
\end{proof}

\begin{prop}
  非自明な木 $T$ の葉の数は
  \[
    2+\sum_{\substack{v\in V(T)\\ \deg_T(v)\geq 3}}\left(\deg_T(v)-2\right)
  \]
  に等しい。
\end{prop}

\begin{proof}
  $n_i$ を次数 $i$ の頂点の数とし、葉の数を $n_1$ とする。木では $|E(T)|=|V(T)|-1$ が成り立つので、
  \[
    \sum_{v\in V(T)}\deg_T(v)=2|E(T)|=2|V(T)|-2
  \]
  である。すなわち
  \[
    n_1+2n_2+\sum_{i\geq3} i n_i
    =
    2n_1+2n_2+\sum_{i\geq3}2n_i-2.
  \]
  これを整理すると
  \[
    n_1=2+\sum_{i\geq3}(i-2)n_i
  \]
  となり、主張を得る。
\end{proof}

\begin{prop}
  グラフ $G$ が木であるための必要十分条件は、$G$ が cycle をもたず、かつ $|E(G)|=|V(G)|-1$ が成り立つことである。
\end{prop}

\begin{proof}
  $G$ が木ならば、定義より cycle をもたない。さらに頂点数に関する帰納法で $|E(G)|=|V(G)|-1$ が従う。実際、非自明な木には葉 $v$ が存在し、$G-v$ も木であるから
  \[
    |E(G)|-1=|E(G-v)|=|V(G-v)|-1=|V(G)|-2.
  \]

  逆に $G$ が cycle をもたず、$|E(G)|=|V(G)|-1$ を満たすとする。$G$ の連結成分を $G_1,\ldots,G_k$ とおくと、各 $G_i$ は木である。よって
  \[
    |E(G)|=\sum_{i=1}^k |E(G_i)|
    =\sum_{i=1}^k \left(|V(G_i)|-1\right)
    =|V(G)|-k.
  \]
  仮定より $k=1$ であるから、$G$ は連結である。したがって $G$ は木である。
\end{proof}

\begin{prop}
  グラフ $G$ が木であるための必要十分条件は、$G$ が連結であり、かつ $|E(G)|=|V(G)|-1$ が成り立つことである。
\end{prop}

\begin{proof}
  $G$ が木ならば、定義より連結であり、前の命題より $|E(G)|=|V(G)|-1$ が成り立つ。

  逆に $G$ が連結で $|E(G)|=|V(G)|-1$ を満たすとする。もし $G$ が cycle をもつならば、その cycle 上の辺を取り除いても連結性は保たれる。この操作を cycle がなくなるまで繰り返すと、$G$ の全域木 $T$ が得られる。しかし $T$ は木なので $|E(T)|=|V(G)|-1$ であり、辺を取り除いたことに矛盾する。よって $G$ は cycle をもたず、木である。
\end{proof}

\end{cor}

\begin{prop}
  頂点集合が $N_n$ である木の個数は $n^{n-2}$ である。

\begin{proof}

\end{proof}

\begin{dfn}
  各辺 $e$ に非負の数 $\mu(e)$ が割り当てられたグラフを \textbf{重み付きグラフ} という。重み付きグラフにおいて、path 上の辺の重みの総和が最小となる path を求める問題を \textbf{最短経路問題} という。
\end{dfn}

\begin{prop}[Dijkstra のアルゴリズム]
  重みが非負である重み付きグラフ $G$ と始点 $u$、終点 $v$ に対し、次の手順で $u$ から $v$ への最短距離を求める。
  \begin{enumerate}[label=(\arabic*)\, ]
    \item 各頂点 $x$ にラベル $(p(x),\ell(x))$ を付ける。初期値は $\ell(u)=0$、$p(u)=*$、$u$ 以外では $\ell(x)=\infty$、$p(x)=*$ とする。
    \item 未確定の頂点のうち $\ell(x)$ が最小のものを選び、そのラベルを確定する。
    \item $x=v$ ならば $\ell(v)$ を出力して終了する。$\ell(v)=\infty$ のときは $u$ から $v$ への path は存在しない。
    \item $x$ に隣接する未確定頂点 $y$ について、$\ell(x)+\mu(xy)<\ell(y)$ ならば、ラベルを $(x,\ell(x)+\mu(xy))$ に更新する。
    \item 終了していなければ (2) に戻る。
  \end{enumerate}
  確定後の $p(x)$ を終点からたどることで、最短 path も復元できる。
\end{prop}

\begin{proof}
  確定した頂点 $x$ について、$\ell(x)$ が $u$ から $x$ への最短距離であることを、確定順に関する帰納法で示す。最初に確定する $u$ では明らかである。次に確定する頂点を $x$ とし、$u$ から $x$ へ $\ell(x)$ より短い path があると仮定する。その path 上で、確定済み頂点から未確定頂点へ初めて移る辺を $yz$ とする。帰納法の仮定と重みの非負性より、$z$ の暫定距離はその path の $z$ までの長さ以下に更新されている。したがって $\ell(z)<\ell(x)$ となり、$\ell(x)$ が未確定頂点の中で最小として選ばれたことに矛盾する。よってアルゴリズムは正しい。
\end{proof}

\begin{dfn}
  重み付き連結グラフ $G$ の全域木 $T$ の重みを
  \[
    \mu(T)=\sum_{e\in E(T)}\mu(e)
  \]
  と定める。$G$ の全域木のうち重みが最小のものを \textbf{最小全域木} という。
\end{dfn}

\begin{prop}[Kruskal のアルゴリズム]
  重み付き連結グラフ $G$ の最小全域木は、次の手順で求められる。
  \begin{enumerate}[label=(\arabic*)\, ]
    \item 選択済みの辺集合を $F=\emptyset$ とする。
    \item まだ選ばれていない辺のうち、$F$ に加えても cycle を作らない重み最小の辺を選び、$F$ に加える。
    \item $|F|=|V(G)|-1$ になれば、$F$ により誘導される全域木を出力する。そうでなければ (2) に戻る。
  \end{enumerate}
\end{prop}

\begin{proof}
  Kruskal のアルゴリズムで得られた全域木を $T$ とし、その辺を選ばれた順に $e_1,\ldots,e_{n-1}$ とする。任意の全域木 $T'$ をとり、$T$ の辺のうち $T'$ に含まれない最初の辺を $e_k$ とする。$T'+e_k$ は cycle を含む。この cycle 上には $T$ に含まれない辺 $e'$ が存在する。$e_k$ が選ばれた時点で $e'$ を選んでも cycle は生じなかったので、Kruskal の選択規則より $\mu(e_k)\leq\mu(e')$ である。したがって $T''=T'+e_k-e'$ は全域木であり、$\mu(T'')\leq\mu(T')$ である。

  この交換を繰り返すと、重みを増やさずに $T'$ を $T$ へ変形できる。よって $\mu(T)\leq\mu(T')$ であり、$T$ は最小全域木である。
\end{proof}
