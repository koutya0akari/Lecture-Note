\begin{tcolorbox}[colback=white,colframe=Turquoise4,,title=公理的確率の存在]
$\Omega : \text{non empty set}$ \par
集合関数 $P$ が次を満たすとき、 $P$ を $\Omega$ 上の（\textbf{公理的}）\textbf{確率}と呼ぶ :
\begin{itemize}
    \item $P(\Omega) = 1$
    \item 任意の $A \subseteq \Omega$ に対して、$P(A) \geq 0$ 
    \item 非交な集合列 $A_1, A_2, \ldots$ に対して、 $P\left( \bigcup_{i=1}^{\infty} A_i \right) = \sum_{i=1}^{\infty} P(A_i)$
\end{itemize}
\end{tcolorbox}

\begin{tcolorbox}[colback=white,colframe=Turquoise4,,title=リーマン積分]
$f \colon [a,b] \to \mathbb{R}$ を区間 $[a,b]$ 上の関数とする。 $f$ が \textbf{リーマン積分可能} であるとは、次の条件を満たすことをいう :
\begin{itemize}
    \item 任意の $\epsilon > 0$ に対して、区間 $[a,b]$ を分割する分割 $P$ が存在し、任意の $P$ に対して、上和 $U(f,P)$ と下和 $L(f,P)$ の差が $\epsilon$ より小さいこと。
\end{itemize}
\end{tcolorbox}

\begin{tcolorbox}[colback=white,colframe=Turquoise4,,title=リーマン積分の欠点1]
    リーマン積分可能な関数を特徴づけることができない。
    \begin{prop}
        $f$ が区間 $[a,b]$ でリーマン積分可能ならば、 $f$ は区間 $[a,b]$ 上で有界。
    \end{prop}
    これは逆が成り立たない。例えば、 $f(x) = \begin{cases} 1 & x \in \mathbb{Q} \\ 0 & x \in \mathbb{R} \setminus \mathbb{Q} \end{cases}$ は区間 $[0,1]$ 上で有界だが、リーマン積分可能ではない。
\end{tcolorbox}

\begin{tcolorbox}[colback=white,colframe=Turquoise4,,title=リーマン積分の欠点2]
    リーマン積分は、項別積分に関して問題がある。\par
    例えば、 $f_n(x) = \begin{cases} 1 & x \in [0,1/n] \\ 0 & x \in (1/n,1] \end{cases}$ とすると、各 $f_n$ は区間 $[0,1]$ 上でリーマン積分可能であり、 $\int_0^1 f_n(x) dx = 1/n$ となる。しかし、 $\lim_{n \to \infty} f_n(x) = 0$ であるため、 $\int_0^1 \lim_{n \to \infty} f_n(x) dx = 0$ となる。つまり、 
    \[\lim_{n \to \infty} \int_0^1 f_n(x) dx \neq \int_0^1 \lim_{n \to \infty} f_n(x) dx\].
\end{tcolorbox}

これらの問題を解決するために、ルベーグ積分が導入される。

\section{区間についての諸性質}

以降、$I$ や $I_n$ は半開区間を表すものとする。
\begin{prop}\label{prop:prop-1}
    \quad $I_1, I_2$ : 半開区間.
    このとき、以下が成り立つ :
    \begin{enumerate}[label=(\roman*)\,,ref=(\roman*)]
        \item \label{itm:prop-1-i} $I_1 \cap I_2$ は半開区間である。
        \item \label{itm:prop-1-ii} $I_1 \setminus I_2$ は直和である。
    \end{enumerate}
\end{prop}

\begin{rem}
    空集合 $\emptyset$ も半開区間とみなすことにする。
\end{rem}
    
\begin{rem}
    集合 $A$ が集合 $B$ と集合 $C$ の\textbf{直和}であるとは、 $A = B \cup C$ かつ $B \cap C = \emptyset$ であることをいう。
\end{rem}

\begin{proof}[\ref{itm:prop-1-i} の証明]
    $I_1 = \emptyset$ または $I_2 = \emptyset$ の場合、$I_1 \cap I_2 = \emptyset$ となるため半開区間である。\par
    次に、$I_1$ と $I_2$ がともに空でない場合を考える。$I_1 = [a,b)$、$I_2 = [c,d)$ とし、一般性を失うことなく $a \leq c$ とする。このとき、区間の終点の位置関係によって以下の3つの場合に分けられる。
    \begin{enumerate}[label=(\arabic*)\,]
        \item $b \leq c$ の場合：$I_1$ と $I_2$ は交わらないため、$I_1 \cap I_2 = \emptyset$ となり半開区間である。
        \item $c < b \leq d$ の場合：$I_1 \cap I_2 = [c,b)$ となり半開区間である。
        \item $d < b$ の場合：$I_2 \subseteq I_1$ となるため、$I_1 \cap I_2 = [c,d)$ となり半開区間である。
    \end{enumerate}
    よって、いずれの場合においても $I_1 \cap I_2$ は半開区間となる。
\end{proof}

\begin{proof}[\ref{itm:prop-1-ii} の証明]
    $I_1 = \emptyset$ の場合は $I_1 \setminus I_2 = \emptyset$ となり直和である。\par
    以下、$I_1 \neq \emptyset$ として $I_1 = [a,b)$ とおく。$I_2 = \emptyset$ の場合は $I_1 \setminus I_2 = [a,b)$ となり直和であるため、$I_2 \neq \emptyset$ として $I_2 = [c,d)$ とおく。
    \begin{enumerate}[label=(\arabic*)\,]
        \item $d \leq a$ または $b \leq c$ の場合：$I_1$ と $I_2$ は交わらないため、$I_1 \setminus I_2 = [a,b)$ となり直和である。
        \item $c \leq a$ かつ $a < d$ の場合：$I_1 \setminus I_2 = [\max(d, a), b)$ となる（$d \geq b$ であれば $\emptyset$）。これは単一の半開区間であり直和である。
        \item $a < c < b$ の場合：$I_1 \setminus I_2 = [a,c) \cup [\min(d, b), b)$ となる（$d \geq b$ であれば後半は $\emptyset$）。互いに素な半開区間の和となるため直和である。
    \end{enumerate}
    以上より、いかなる場合も $I_1 \setminus I_2$ は直和として表される。
\end{proof}

\begin{dfn}
    $I = [a,b)$ を半開区間とする。 $I$ の\textbf{長さ}を $|I| := b - a$ と定義する。
\end{dfn}

\begin{prop}\label{prop:prop-2}
    $I = [a,b)$ を半開区間とする。このとき、以下が成り立つ :
    \begin{enumerate}[label=(\roman*)\,,ref=(\roman*)]
        \item \label{itm:prop-2-i} $I \subseteq \bigcup_{i=1}^{n} I_i$ ならば、 $|I| \leq \sum_{i=1}^{n} |I_i|$ である。
        \item \label{itm:prop-2-ii} $I_1, I_2, \ldots, I_n$ が互いに素な半開区間で、 $\bigcup_{i=1}^{n} I_i \subseteq I$ であるならば、 $\sum_{i=1}^{n} |I_i| \leq |I|$ である。
        \item \label{itm:prop-2-iii} $I_1, I_2, \ldots, I_n$ が互いに素な半開区間で、 $I = \bigcup_{i=1}^{n} I_i$ であるならば、 $|I| = \sum_{i=1}^{n} |I_i|$ である。
    \end{enumerate}
    ここで、半開区間 $I_1, I_2, \ldots, I_n$ が\textbf{互いに素}であるとは、任意の $i \neq j$ に対して、 $I_i \cap I_j = \emptyset$ であることをいう。
\end{prop}

\begin{proof}[\ref{itm:prop-2-i} の証明]
区間をそれぞれ $I=[a,b)$、および $I_i=[a_i,b_i) \quad (i=1,2,\cdots,n)$ とおく。

$s$ を任意の自然数とする。
$a \leq \frac{r}{s} < b$、すなわち $\frac{r}{s} \in I$ を満たす有理数 $\frac{r}{s}$ （$r$ は整数）の個数を $N$ とする。同様に、$a_i \leq \frac{r}{s} < b_i$、すなわち $\frac{r}{s} \in I_i$ を満たす $\frac{r}{s}$ の個数を $N_i$ とする。

$\frac{r}{s} \in I$ であるとき、仮定 $I \subseteq \bigcup_{i=1}^{n} I_i$ により、その点は必ずいずれかの $I_i$ に属している。したがって、各区間に属する点の個数の間には次の不等式が成り立つ:
\[
 N \leq N_1 + N_2 + \cdots + N_n
\]

ここで、実数 $a, b$ と自然数 $s$ に対して、
\[
 \frac{p-1}{s} < a \leq \frac{p}{s}, \qquad \frac{q}{s} < b \leq \frac{q+1}{s}
\]
を満たす整数 $p, q$ を一意に定めることができる。これを変形すると $sa < p \leq sa + 1$ および $sb - 1 \leq q < sb$ となり、$a \leq \frac{r}{s} < b$ を満たす整数 $r$ は $p \leq r \leq q$ の範囲の整数であることがわかる。
この条件を満たす整数の個数 $N$ は $N = q - p + 1$ であるため、先ほどの $p, q$ の不等式によって
\[
 s(b-a) - 1 < N < s(b-a) + 1
\]
と評価できる。

各 $I_i$ についても同様に不等式を立てることで、
\[
 s(b_i - a_i) - 1 < N_i < s(b_i - a_i) + 1 \quad (i=1,2,\cdots,n)
\]
が成り立つ。

これらの不等式を組み合わせることで、$N$ および各 $N_i$ の和についての評価は次のように展開できる。
\[
 s(b-a) - 1 < N \leq \sum_{i=1}^{n} N_i < \sum_{i=1}^{n} \{s(b_i - a_i) + 1\}
\]
両辺を $s$ で割って整理すると、
\[
 b - a - \frac{1}{s} < \sum_{i=1}^{n} \left( (b_i - a_i) + \frac{1}{s} \right)
\]
となる。ここで $s \to +\infty$ の極限をとれば、$\frac{1}{s}$ に依存する項は $0$ に収束するため、
\[
 b - a \leq \sum_{i=1}^{n} (b_i - a_i)
\]
が導かれる。したがって、区間の長さの定義より
\[
 |I| \leq \sum_{i=1}^{n} |I_i|
\]
が示された。
\end{proof}

\begin{rem}
\ref{itm:prop-2-i} は次のようにして簡単に証明できる : 
必要に応じ番号を付けかえた上で、
\[
 a_1\leq a<b_1,\qquad
 a_n<b\leq b_n,
\]
\[
 a_{p+1}<b_p\leq b_{p+1}
 \quad (p=1,2,\cdots,n-1)
\]
であると仮定しても一般性を失わないため、
\[
 b-a
 \leq b_n-a_1
 =
 (b_1-a_1)+\sum_{p=1}^{n-1}(b_{p+1}-b_p)
 \leq
 \sum_{p=1}^{n}(b_p-a_p).
\]
一般に $\R^n$ で\ref{itm:prop-2-i} に類似の定理をこれと同様な方法で証明しようとすると、区間の順序関係を利用することができないため、同様の証明は困難である。最初に提示した証明法なら、そのまま $\R^n$ の場合にも適用できる。
この証明は J. von Neumann によるものである。
尚、これは次の定理 \ref{itm:prop-2-ii} にもあてはまる。
\end{rem}

\begin{proof}[\ref{itm:prop-2-ii} の証明]
\ref{itm:prop-2-i} の証明で用いた記号および各個数 $N, N_i$ の評価を引き続き使用する。
今回は $I_1, \ldots, I_n$ が互いに素であるため、特定の点 $\frac{r}{s}$ が複数の区間 $I_i$ に同時に属することは存在しない。さらに、仮定より $\bigcup_{i=1}^{n} I_i \subseteq I$ となっているため、区間 $I_i$ に属する有理数 $\frac{r}{s}$ は必ず区間 $I$ にも属している。
したがって、各区間に属する点の個数の間の不等式は、今回は逆向きの包含関係に注目でき、
\[
 \sum_{i=1}^{n} N_i \leq N
\]
という形に変わる。

この不等式に対し、各 $N_i, N$ についての評価を代入することで、
\[
 \sum_{i=1}^{n} \left( b_i - a_i - \frac{1}{s} \right) < \sum_{i=1}^{n} \frac{N_i}{s} \leq \frac{N}{s} < (b-a) + \frac{1}{s}
\]
が得られる。再びここで $s \to +\infty$ の極限をとることで、$\frac{1}{s}$ に依存する項が消え、
\[
 \sum_{i=1}^{n} (b_i - a_i) \leq (b - a)
\]
となる。以上により、区間の長さに関する不等式
\[
 \sum_{i=1}^{n} |I_i| \leq |I|
\]
が成立する。
\end{proof}

\begin{proof}[\ref{itm:prop-2-iii} の証明]
仮定より $I = \bigcup_{i=1}^{n} I_i$ であるため、これは $I \subseteq \bigcup_{i=1}^{n} I_i$ かつ $\bigcup_{i=1}^{n} I_i \subseteq I$ の双方の包含関係が成り立つことを意味する。

前半の包含関係 $I \subseteq \bigcup_{i=1}^{n} I_i$ に対して \ref{itm:prop-2-i} の結果を適用することで、
\[
 |I| \leq \sum_{i=1}^{n} |I_i|
\]
が従う。

一方で、後半の包含関係 $\bigcup_{i=1}^{n} I_i \subseteq I$ に対して、さらに $I_i$ が互いに素であるという仮定を用いて \ref{itm:prop-2-ii} の結果を適用することで、
\[
 |I| \geq \sum_{i=1}^{n} |I_i|
\]
を得る。

これらの相反する不等式が同時に成り立つため、最終的に
\[
 |I| = \sum_{i=1}^{n} |I_i|
\]
であることが示される。
\end{proof}

\begin{prop}\label{prop:prop-3}
    \quad $I, I_1, I_2, \ldots$ : 半開区間\\
    このとき、以下が成り立つ :
    \begin{enumerate}[label=(\roman*)\,,ref=(\roman*)]
        \item \label{itm:prop-3-i} $I\subset \bigcup_{n=1}^{\infty}I_n$ ならば $|I|\leq \sum_{n=1}^{\infty}|I_n|$.
        \item \label{itm:prop-3-ii} $I_1,I_2,\cdots,I_n,\cdots$ が互いに素な半閉区間の列で $I=\bigcup_{n=1}^{\infty}I_n$ ならば $|I|=\sum_{n=1}^{\infty}|I_n|$.
    \end{enumerate}
\end{prop}

\begin{proof}[\ref{itm:prop-3-i} の証明]
$\sum_{n=1}^{\infty}|I_n|$ が発散する場合、不等式の右辺は無限大となるため自明に成り立つ。したがって、級数が正の有限な値に収束する場合（$\sum_{n=1}^{\infty}|I_n|<+\infty$）についてのみ証明すれば十分である。

各半開区間を以下のように定める。
\[
 I=[a,b),\qquad
 I_n=[a_n,b_n)\quad (n=1,2,\cdots)
\]
ここで、任意の正の実数 $\varepsilon$ に対して、十分小さな正の実数 $\eta$ および $\eta_n$ を次のように選ぶ。
\[
 \eta < \min\left\{ \frac{\varepsilon}{2},\, \frac{b-a}{2} \right\}, \qquad
 \eta_n < \frac{\varepsilon}{2^{n+1}}
\]
これらの正数を用いて、区間 $I$ および $I_n$ を少し狭めたり広げたりした新たな区間を導入する。
\[
 J=[a,b-\eta), \qquad J^{a}=[a,b-\eta]
\]
\[
 J_n=[a_n-\eta_n,b_n), \qquad J_n^{i}=(a_n-\eta_n,b_n)
\]
定義より、明らかに $J \subset J^{a} \subset I$ が成り立ち、また各 $n$ について $I_n \subset J_n^{i}$ が成り立つ。
これらを仮定 $I \subset \bigcup_{n=1}^{\infty}I_n$ と組み合わせることで、有界閉区間 $J^{a}$ は次のような開区間の和集合で覆われることがわかる。
\[
 J^{a} \subset I \subset \bigcup_{n=1}^{\infty}I_n \subset \bigcup_{n=1}^{\infty}J_n^{i}
\]
ここで $J^{a}$ は有界閉集合（コンパクト集合）であり、各 $J_n^{i}$ は開集合であるため、被覆定理を適用できる。

ハイネ・ボレルの被覆定理（Borel--Lebesgue の被覆定理）によれば、有界閉集合を覆う開被覆からは有限個の部分被覆を選び出すことができる。したがって、要素数 $k$ の有限の部分集合
\[
 \{n(1), n(2), \ldots, n(k)\} \subset \mathbb{N}
\]
を適切に選ぶことで、次のように $J^{a}$ を有限個の集合で覆うことができる。
\[
 J^{a} \subset J_{n(1)}^{i} \cup J_{n(2)}^{i} \cup \cdots \cup J_{n(k)}^{i}
\]
これを、先ほどの部分集合関係により少し狭めた半開区間の言葉で書き直すと、
\[
 J \subset J_{n(1)} \cup J_{n(2)} \cup \cdots \cup J_{n(k)}
\]
となる。これは有限個の半開区間による被覆であるため、\ref{itm:prop-2-i} の結果を適用することができ、
\[
 |J| \leq |J_{n(1)}| + |J_{n(2)}| + \cdots + |J_{n(k)}| \leq \sum_{n=1}^{\infty}|J_n|
\]
という不等式を得る。

一方で、各々の区間の選び方により、長さに以下の通りの差が存在する。
\[
 0 < |I|-|J| = \eta < \frac{\varepsilon}{2}, \qquad
 0 < |J_n|-|I_n| = \eta_n < \frac{\varepsilon}{2^{n+1}}
\]
これらの差分の評価を、上の有限和の長さに適用していくことで、元の区間の長さ $|I|$ についての全体的な評価を構成する。
\[
 |I| < |J| + \frac{\varepsilon}{2} \leq \left( \sum_{n=1}^{\infty}|J_n| \right) + \frac{\varepsilon}{2}
 < \sum_{n=1}^{\infty}\left( |I_n| + \frac{\varepsilon}{2^{n+1}} \right) + \frac{\varepsilon}{2}
\]
もっとも右辺にある級数の和を展開して整理する。等比数列の和の公式より $\sum_{n=1}^{\infty} \frac{1}{2^{n+1}} = \frac{1}{2}$ であることを用いれば、
\[
 \sum_{n=1}^{\infty}|I_n| + \varepsilon \sum_{n=1}^{\infty}\frac{1}{2^{n+1}} + \frac{\varepsilon}{2}
 = \sum_{n=1}^{\infty}|I_n| + \frac{\varepsilon}{2} + \frac{\varepsilon}{2}
 = \sum_{n=1}^{\infty}|I_n| + \varepsilon
\]
となる。
以上の通り、任意の正の実数 $\varepsilon$ に対して、
\[
 |I| < \sum_{n=1}^{\infty}|I_n| + \varepsilon
\]
が導かれた。ここで $\varepsilon \to +0$ と極限をとることで、この任意性が外れ、最終的な性質
\[
 |I| \leq \sum_{n=1}^{\infty}|I_n|
\]
が示されることになる。
\end{proof}

\section{外測度の定義}

すべての部分集合に対して、外測度という「大きさ」の相当するような値を定めることを目標にする。

外測度を定義するために、いくつか復習をする。

\begin{dfn}
    \quad $A \subseteq \mathbb{R}, A \neq \emptyset$.
    \begin{enumerate}[label=(\roman*)\,,ref=(\roman*)]
        \item $m \in \R$ が $A$ の\textbf{最大値} であるとは、 $m \in A$ かつ 任意の $x \in A$ に対して $x \leq m$ であることをいう。
        \item $m \in \R$ が $A$ の\textbf{最小値} であるとは、 $m \in A$ かつ 任意の $x \in A$ に対して $m \leq x$ であることをいう。
    \end{enumerate}
\end{dfn}

\begin{rem}
    一般に、集合 $A$ が最大値や最小値を持つとは限らない。例えば、開区間 $(0,1)$ は最大値も最小値も持たない。
\end{rem}

\begin{dfn}
    \quad $A \subseteq \mathbb{R}, A \neq \emptyset$.
    \begin{enumerate}[label=(\roman*)\,,ref=(\roman*)]
        \item $m \in \R$ が $A$ の\textbf{上界} であるとは、任意の $x \in A$ に対して $x \leq m$ であることをいう。
        \item $m \in \R$ が $A$ の\textbf{下界} であるとは、任意の $x \in A$ に対して $m \leq x$ であることをいう。
    \end{enumerate}
\end{dfn}

\begin{axiom}
    \quad $A \subseteq \mathbb{R}$.
    \begin{enumerate}[label=(\roman*)\,,ref=(\roman*)]
        \item $A$ が上界を持つならば、 $A$ は最小値を持つ。
        \item $A$ が下界を持つならば、 $A$ は最大値を持つ。
    \end{enumerate}
\end{axiom}

この公理により、以下の定義が well-defined であることが保証される。
